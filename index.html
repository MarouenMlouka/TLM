function ajouterPoints(index) { var motCle = prompt("Pour ajouter des points, veuillez saisir le mot-clé :"); if (motCle === "TLM") { var pointsAjoutes = parseInt(prompt("Combien de points souhaitez-vous ajouter ?")); console.log("Points à ajouter :", pointsAjoutes); // Débogage if (!isNaN(pointsAjoutes) && pointsAjoutes > 0) { const pointsRef = db.ref('points/' + index); pointsRef.transaction((currentPoints) => { console.log("Points actuels :", currentPoints); // Débogage return (currentPoints || 0) + pointsAjoutes; }, (error, committed, snapshot) => { if (error) { alert("Une erreur est survenue lors de l'ajout des points : " + error.message); } else if (committed) { alert(pointsAjoutes + " points ajoutés avec succès !"); } }); } else { alert("Veuillez saisir un nombre valide de points."); } } else { alert("Mot-clé incorrect. Vous n'êtes pas autorisé à ajouter des points."); } }
